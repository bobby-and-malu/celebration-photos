    <script>

        // 1. PASTE YOUR GOOGLE WEB APP ENDPOINT URL HERE

        const WEB_APP_URL = "YOUR_WEB_APP_URL_HERE";


        // Security constants enforced on client-side

        const MAX_FILE_SIZE_MB = 10; 

        const MAX_FILE_SIZE_BYTES = MAX_FILE_SIZE_MB * 1024 * 1024;


        const photoInput = document.getElementById('photoInput');

        const fileNameDiv = document.getElementById('fileName');

        const uploadForm = document.getElementById('uploadForm');

        const submitBtn = document.getElementById('submitBtn');

        const statusDiv = document.getElementById('status');


        // Dynamically updates selected file status + basic client checks

        photoInput.addEventListener('change', function() {

            if (this.files && this.files.length > 0) {

                const file = this.files[0];

                

                // Security Check #1: Validate it's an image

                if (!file.type.startsWith('image/')) {

                    statusDiv.className = "error";

                    statusDiv.textContent = "❌ Please select an image file (PNG, JPG, HEIC).";

                    this.value = ''; // reset selection

                    fileNameDiv.textContent = "No image chosen yet";

                    return;

                }

                

                // Security Check #2: Enforce size limit before converting to string

                if (file.size > MAX_FILE_SIZE_BYTES) {

                    statusDiv.className = "error";

                    statusDiv.textContent = `❌ File is too large. Maximum size allowed is ${MAX_FILE_SIZE_MB}MB.`;

                    this.value = ''; // reset selection

                    fileNameDiv.textContent = "No image chosen yet";

                    return;

                }


                statusDiv.textContent = ""; // Clear existing error messages if file passes validation

                fileNameDiv.textContent = `${file.name} (${(file.size / (1024 * 1024)).toFixed(2)} MB)`;

            } else {

                fileNameDiv.textContent = "No image chosen yet";

            }

        });


        // Safe pipeline processing form submissions

        uploadForm.addEventListener('submit', function(e) {

            e.preventDefault();

            

            const file = photoInput.files[0];

            if (!file) return;


            // Enter uploading UI visual state

            submitBtn.disabled = true;

            submitBtn.textContent = "Uploading Your Memory... Please Wait";

            statusDiv.textContent = "";


            const reader = new FileReader();

            reader.readAsDataURL(file);

            

            reader.onload = function() {

                const base64String = reader.result.split(',')[1];

                

                const payload = {

                    base64: base64String,

                    type: file.type,

                    filename: file.name

                };


                fetch(WEB_APP_URL, {

                    method: 'POST',

                    body: JSON.stringify(payload)

                })

                .then(response => {

                    if (!response.ok) throw new Error("Network issues encountered.");

                    return response.json();

                })

                .then(data => {

                    if (data.status === 'success') {

                        statusDiv.className = "success";

                        statusDiv.textContent = "Uploaded successfully! Thank you for sharing in our day”;

                        uploadForm.reset();

                        fileNameDiv.textContent = "No image chosen yet";

                    } else {

                        throw new Error(data.message || "Upload failed on server.");

                    }

                })

                .catch(error => {

                    console.error("Transmission Error: ", error);

                    statusDiv.className = "error";

                    statusDiv.textContent = "❌ Upload failed. Please check your connection or try another picture.";

                })

                .finally(() => {

                    submitBtn.disabled = false;

                    submitBtn.textContent = "Upload Photo";

                });

            };

        });
